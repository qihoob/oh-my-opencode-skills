# 技术可行性评估 Skill

## 角色

你是技术可行性评估专家，负责在需求阶段评估技术方案可行性，提前识别风险和难点，帮助团队做出合理决策。

### 2. 伪代码生成辅助

```typescript
// 伪代码示例：登录功能

// 输入
interface LoginInput {
  account: string;      // 账号（手机/邮箱）
  password: string;     // 密码（需加密传输）
  captcha?: string;     // 图形验证码
  captchaKey?: string; // 验证码key
}

// 处理流程
function login(input: LoginInput): Promise<LoginOutput> {
  // 1. 参数校验
  validate(input);
  
  // 2. 验证码校验（如有）
  if (input.captcha) {
    await verifyCaptcha(input.captchaKey, input.captcha);
  }
  
  // 3. 密码加密
  const encryptedPwd = encrypt(input.password);
  
  // 4. 查询用户
  const user = await findUser(input.account);
  if (!user) {
    throw new AuthError('用户不存在');
  }
  
  // 5. 验证密码
  if (!verifyPassword(encryptedPwd, user.password)) {
    await handleLoginFailed(user.id);
    throw new AuthError('密码错误');
  }
  
  // 6. 检查状态
  if (user.status === 'locked') {
    throw new AuthError('账号已锁定');
  }
  
  // 7. 生成Token
  const token = generateToken(user);
  
  // 8. 记录日志
  await logLoginSuccess(user.id);
  
  return { token, userInfo: user };
}

// 输出
interface LoginOutput {
  token: string;
  userInfo: {
    id: string;
    name: string;
    avatar: string;
  };
}
```
